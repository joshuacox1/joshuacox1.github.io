Coin combinations

-- Returns all the ways to make `n` pence with coin denominations from `coins`.
-- If `coins` is sorted ascending or descending, the resulting lists
-- will be sorted ascending or descending respectively.
-- * `n` should be a non-negative integer.
-- * `coins` should be a list of unique positive integers.
coinCombs :: Int -> [Int] -> [[Int]]
coinCombs n coins =
  case coins of
    [] -> if n == 0 then [[]] else []
    c:cs ->
      let withC = if n >= c then map (c:) (coinCombs (n-c) coins) else [] in
      let withoutC = coinCombs n cs in
      withC ++ withoutC
